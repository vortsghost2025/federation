CORE METHODS
============

advance_turn() -> Dict
  Advance to next turn
  Auto-updates game phase based on turn count
  Returns: {success, new_turn, game_phase}

record_action(action_type, description, outcome,
              morale_delta=0.0, stability_delta=0.0,
              treasury_delta=0) -> Dict
  Record player action and update state
  Returns: {success, current_morale, current_stability, current_treasury}

set_victory_condition(victory_type: VictoryType) -> Dict
  Mark federation as won
  Returns: {success, game_won, victory_type, final_turn, final_morale}

set_defeat_condition(reason: str) -> Dict
  Mark federation as defeated
  Returns: {success, game_lost, defeat_reason, final_turn}

get_game_summary() -> Dict
  Comprehensive federation status report
  Returns: full state snapshot with all metrics

get_statistics() -> Dict
  Detailed gameplay statistics
  Returns: {military, diplomacy, resources, population}

reset_game() -> Dict
  Clear all state, start fresh
  Returns: {success, previous_game_stats}

save_game(filepath: str) -> Dict
  Serialize game state to JSON
  Returns: {success, filepath, file_size_bytes, save_count}

load_game(filepath: str) -> Dict
  Restore game state from JSON file
  Returns: {success, turn_restored, game_phase, federation_morale}

validate_state() -> Dict
  Check state integrity and consistency
  Returns: {validation_passed, issues_found, issues, state_hash}

USAGE EXAMPLE
=============

from federation_game_state import GameState, VictoryType

# Initialize
game = GameState()

# Play
for turn in range(100):
    game.advance_turn()
    game.record_action(
        action_type="military",
        description="Battle won",
        outcome="Victory",
        morale_delta=0.1,
        treasury_delta=200
    )

# Save
game.save_game("my_game.json")

# Load later
game2 = GameState()
game2.load_game("my_game.json")

# Check status
summary = game2.get_game_summary()
print(f"Turn {summary['current_turn']}, Phase: {summary['game_phase']}")
